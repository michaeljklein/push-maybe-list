partMaybes :: [Maybe a] -> [Maybe [Maybe a]] Source #

The "natural" partition of a list of Maybe's: we group Nothing's and Just's and the outermost Maybe marks which is contained.

unMaybeParts :: [Maybe [Maybe a]] -> [[Maybe a]] Source #

Convert the "natural" partition into a partition that we can more easily push through the innermost Maybe: replace Nothing's with an empty lists and drop Just's.

partMaybe :: [Maybe a] -> [[Maybe a]] Source #

unMaybeParts . partMaybes

dartMaybe :: [[Maybe a]] -> [Maybe [a]] Source #

Reassociate the list and Maybe

pushMaybe :: [Maybe a] -> [Maybe [a]] Source #

dartMaybe . partMaybe
